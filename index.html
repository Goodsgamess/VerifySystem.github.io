// Requires "node-fetch" in package.json const fetch = require('node-fetch'); exports.handler = async function(event) { try { const { webhook, payload } = JSON.parse(event.body); await fetch(webhook, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) }); return { statusCode: 200, body: 'Success' }; } catch (err) { console.error('Webhook proxy error:', err); return { statusCode: 500, body: 'Error' }; } };
